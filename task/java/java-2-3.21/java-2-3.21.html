public class Exercise1 {

public static void main(String [] args) {

int m = 0;

System.out.println("m++="+(m++));

//先引用再+1，即（m++）=0，m=m+1=1

System.out.println("++m="+(++m));

//先+1再引用，即m=m+1=2,（++m）=2

int n = 10;

System.out.println("m--="+(m--));

//先引用再-1，即(m--)=2,m=m-1=1

System.out.println("--m="+(--m));

//先-1再引用，即m=m-1=0,(--m)=0

boolean x;

x = (5>3) && (4==6);

System.out.println("x="+x);

//(5>3)为true，（4==6）为false，所以x为false

m = m%2;

System.out.println("result="+m+1);

//m = 0%2=0,"+"连接0和1两个字符，输出result=01

int y = m*m+2*m-1;

System.out.println("m="+m+",y="+y);

//m=0，y=0*0+2*0-1=-1

 

}

 

}

解析：

 

public class Exercise2 {

public static void main (String [] args) {

char a = 'h';

//从'h'判断为char字符型

byte b = 6;

int i = 200;

long n = 567L;

//由数字后的L判断为long整数型

float f = 98.99f;

//由数字后的f判断为 float浮点型

double d = 4.7788;

int aa = a+i;

//char字符型a被隐式转换为int整数型a，根据ASCII码表，int a = 104;

//所以int aa = 104+200=304

long nn = n-aa;

//int整数型a被隐式转换为long整数型a，即long a = 304L

//所以long nn = 567L-304L=263L

float ff = (float)(b*d);

//强制转换，由前面为float判断为flaot浮点型

//(b*d)为double双精度浮点数，被强制转换为float单精度浮点数

//所以float ff = 28.6728

double dd = ff/aa+d;

//计算为dd=4.873118419694901

System.out.println("aa="+aa);

System.out.println("nn="+nn);

System.out.println("ff="+ff);

System.out.println("dd="+dd);

}

 

}
